22 auto controller::init() -> bool {
23 m_shard_data.reserve(m_opts.m_shard_endpoints.size());
24 for(size_t i{0}; i < m_opts.m_shard_endpoints.size(); i++) {
25 const auto& shard = m_opts.m_shard_endpoints[i];
26 m_logger->info("Connecting to",
27 shard.first,
28 ":",
29 shard.second,
30 "...");
31 auto sock = std::make_unique<network::tcp_socket>();
32 if(!sock->connect(shard)) {
33 m_logger->error("failed to connect");
34 return false;
35 }
36
37 const auto peer_id = m_shard_network.add(std::move(sock));
38 const auto& shard_range = m_opts.m_shard_ranges[i];
39 m_shard_data.push_back(shard_info{shard_range, peer_id});
40
41 m_logger->info("done");
42 }
43
44 auto rng = std::default_random_engine();
45 rng.seed(m_sentinel_id);
46
47 // Shuffle the shards list to spread load between shards
48 std::shuffle(m_shard_data.begin(), m_shard_data.end(), rng);
49
50 auto rpc_server = std::make_unique<cbdc::rpc::tcp_server<
51 cbdc::rpc::blocking_server<request, response>>>(
52 m_opts.m_sentinel_endpoints[m_sentinel_id]);
53 if(!rpc_server->init()) {
54 m_logger->error("Failed to start sentinel RPC server");
55 return false;
56 }
57
58 m_rpc_server = std::make_unique<decltype(m_rpc_server)::element_type>(
59 this,
60 std::move(rpc_server));
61
62 return true;
63 }

65 auto controller::execute_transaction(transaction::full_tx tx)
66 -> std::optional<cbdc::sentinel::response> {
67 const auto res = transaction::validation::check_tx(tx);
68 tx_status status{tx_status::pending};
69 if(res.has_value()) {
70 status = tx_status::static_invalid;
71 }
72
73 auto tx_id = transaction::tx_id(tx);
74
75 if(!res.has_value()) {
76 m_logger->debug("Accepted tx:", cbdc::to_string(tx_id));
77 } else {
78 m_logger->debug("Rejected tx:", cbdc::to_string(tx_id));
79 }
80
81 // Only forward transactions that are valid
82 if(!res.has_value()) {
83 send_transaction(tx);
84 }
85
86 return response{status, res};
87 }

89 void controller::send_transaction(const transaction::full_tx& tx) {
90 const auto compact_tx = cbdc::transaction::compact_tx(tx);
91 auto ctx_pkt = std::make_shared<cbdc::buffer>();
92 auto ctx_ser = cbdc::buffer_serializer(*ctx_pkt);
93 ctx_ser << compact_tx;
94
95 auto inputs_sent = std::unordered_set<size_t>();
96 for(const auto& [range, pid] : m_shard_data) {
97 if(inputs_sent.size() == compact_tx.m_inputs.size()) {
98 break;
99 }
100 if(!m_shard_network.connected(pid)) {
101 continue;
102 }
103 auto should_send = false;
104 for(size_t i = 0; i < compact_tx.m_inputs.size(); i++) {
105 if(inputs_sent.find(i) != inputs_sent.end()) {
106 continue;
107 }
108 if(!config::hash_in_shard_range(range,
109 compact_tx.m_inputs[i])) {
110 continue;
111 }
112 inputs_sent.insert(i);
113 should_send = true;
114 }
115 if(should_send) {
116 m_shard_network.send(ctx_pkt, pid);
117 }
118 }
119 }

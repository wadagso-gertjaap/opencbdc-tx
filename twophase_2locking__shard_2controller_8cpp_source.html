51 auto controller::init() -> bool {
52 auto params = nuraft::raft_params();
53 params.election_timeout_lower_bound_
54 = static_cast<int>(m_opts.m_election_timeout_lower);
55 params.election_timeout_upper_bound_
56 = static_cast<int>(m_opts.m_election_timeout_upper);
57 params.heart_beat_interval_ = static_cast<int>(m_opts.m_heartbeat);
58 params.snapshot_distance_ = 0; // TODO: implement snapshots
59 params.max_append_size_ = static_cast<int>(m_opts.m_raft_max_batch);
60
61 if(!m_raft_serv->init(params)) {
62 m_logger->error("Failed to initialize raft server");
63 return false;
64 }
65
66 if(!m_raft_serv->build_cluster(
67 m_opts.m_locking_shard_raft_endpoints[m_shard_id])) {
68 m_logger->error("Failed to build raft cluster");
69 return false;
70 }
71
72 auto status_rpc_server = std::make_unique<
73 cbdc::rpc::blocking_tcp_server<rpc::status_request,
74 rpc::status_response>>(
75 m_opts.m_locking_shard_readonly_endpoints[m_shard_id][m_node_id]);
76 if(!status_rpc_server->init()) {
77 m_logger->error("Failed to start status RPC server");
78 return false;
79 }
80
81 m_status_server
82 = std::make_unique<decltype(m_status_server)::element_type>(
83 m_shard,
84 std::move(status_rpc_server));
85
86 return true;
87 }

89 auto controller::raft_callback(nuraft::cb_func::Type type,
90 nuraft::cb_func::Param* /* param */)
91 -> nuraft::cb_func::ReturnCode {
92 if(type == nuraft::cb_func::Type::BecomeFollower) {
93 m_logger->warn("Became follower, stopping listener");
94 m_server.reset();
95 return nuraft::cb_func::ReturnCode::Ok;
96 }
97 if(type == nuraft::cb_func::Type::BecomeLeader) {
98 m_logger->warn("Became leader, starting listener");
99 m_server = std::make_unique<decltype(m_server)::element_type>(
100 m_opts.m_locking_shard_endpoints[m_shard_id][m_node_id]);
101 m_server->register_raft_node(m_raft_serv);
102 if(!m_server->init()) {
103 m_logger->fatal("Couldn't start message handler server");
104 }
105 }
106 return nuraft::cb_func::ReturnCode::Ok;
107 }

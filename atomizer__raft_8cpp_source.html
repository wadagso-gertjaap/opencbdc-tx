52 void atomizer_raft::tx_notify(tx_notify_request&& notif) {
53 auto it = m_txs.find(notif.m_tx);
54 if(it != m_txs.end()) {
55 for(auto n : notif.m_attestations) {
56 auto p = std::make_pair(n, notif.m_block_height);
57 auto n_it = it->second.find(p);
58 if((n_it != it->second.end()
59 && n_it->second < notif.m_block_height)
60 || n_it == it->second.end()) {
61 it->second.insert(std::move(p));
62 }
63 }
64 } else {
65 auto attestations = attestation_set();
66 attestations.reserve(notif.m_attestations.size());
67 for(auto n : notif.m_attestations) {
68 attestations.insert(std::make_pair(n, notif.m_block_height));
69 }
70 it = m_txs
71 .insert(std::make_pair(std::move(notif.m_tx),
72 std::move(attestations)))
73 .first;
74 }
75
76 // TODO: handle notifications that never spill over due to lack of
77 // attestations
78 if(it->second.size() == it->first.m_inputs.size()) {
79 auto agg = aggregate_tx_notification();
80 auto tx = m_txs.extract(it);
81 agg.m_tx = std::move(tx.key());
82 uint64_t oldest{0};
83 for(const auto& att : tx.mapped()) {
84 if(oldest == 0 || att.second < oldest) {
85 oldest = att.second;
86 }
87 }
88 agg.m_oldest_attestation = oldest;
89 {
90 std::lock_guard<std::mutex> l(m_complete_mut);
91 m_complete_txs.push_back(std::move(agg));
92 }
93 }
94 }

96 auto atomizer_raft::send_complete_txs(const raft::callback_type& result_fn)
97 -> bool {
98 auto atns = aggregate_tx_notify_request();
99 {
100 std::lock_guard<std::mutex> l(m_complete_mut);
101 std::swap(atns.m_agg_txs, m_complete_txs);
102 }
103 if(atns.m_agg_txs.empty()) {
104 return false;
105 }
106 return make_request(atns, result_fn);
107 }

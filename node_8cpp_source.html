38 auto node::init(const nuraft::raft_params& raft_params) -> bool {
39 // Set these explicitly as they depend on the structure of this class
40 auto params = raft_params;
41 if(m_blocking) {
42 params.return_method_ = nuraft::raft_params::blocking;
43 } else {
44 params.return_method_ = nuraft::raft_params::async_handler;
45 }
46 params.auto_forwarding_ = false;
47
48 m_raft_instance = m_launcher.init(m_sm,
49 m_smgr,
50 m_raft_logger,
51 m_port,
52 m_asio_opt,
53 params,
54 m_init_opts);
55
56 if(!m_raft_instance) {
57 std::cerr << "Failed to initialize raft launcher" << std::endl;

59 }
60

63 while(!m_raft_instance->is_initialized()) {
64 std::cout << "." << std::flush;
65 std::this_thread::sleep_for(wait_time);

67 std::cout << std::endl;

72 auto node::add_cluster_nodes(
73 const std::vector<network::endpoint_t>& raft_servers) const -> bool {
74 static constexpr auto sleep_time = std::chrono::milliseconds(100);
75
76 auto srvs = std::vector<std::pair<int, std::string>>();
77 for(size_t i{0}; i < raft_servers.size(); i++) {
78 if(i != static_cast<size_t>(m_node_id)) {
79 const auto& ep = raft_servers[i];
80 auto ep_str = ep.first + ":" + std::to_string(ep.second);
81 srvs.emplace_back(std::make_pair(i + 1, ep_str));
82 }
83 }
84
85 for(const auto& srv_data : srvs) {
86 nuraft::srv_config srv(srv_data.first, srv_data.second);
87
88 for(;;) {
89 std::cout << "Adding raft server: " << srv.get_id() << ", "
90 << srv.get_endpoint() << std::flush;
91
92 auto ret = m_raft_instance->add_srv(srv);
93 if(!ret->get_accepted()
94 && ret->get_result_code()
95 != nuraft::cmd_result_code::SERVER_IS_JOINING) {
96 std::cout << "Failed to add raft server: " << srv.get_id()
97 << ", " << srv.get_endpoint()
98 << ", error: " << ret->get_result_str()
99 << std::endl;
100 return false;
101 }
102
103 nuraft::ptr<nuraft::srv_config> srv_conf;
104 int attempts{0};
105 const auto max_retries = 50;
106 do {
107 srv_conf = m_raft_instance->get_srv_config(srv_data.first);
108 std::cout << "." << std::flush;
109 attempts++;
110 std::this_thread::sleep_for(sleep_time);
111 } while(!srv_conf && attempts < max_retries);
112
113 if(!srv_conf) {
114 std::cout << "timed out" << std::endl;
115 if(!m_wait_for_followers) {
116 return false;
117 }
118 } else {
119 break;
120 }
121 }
122
123 std::cout << "done" << std::endl;
124 }
125
126 return true;
127 }

129 auto
130 node::build_cluster(const std::vector<network::endpoint_t>& raft_servers)
131 -> bool {
132 std::vector<nuraft::ptr<nuraft::srv_config>> srv_configs;
133 m_raft_instance->get_srv_config_all(srv_configs);
134
135 static constexpr auto sleep_time = std::chrono::milliseconds(100);
136
137 if(srv_configs.size() < raft_servers.size()) {
138 if(m_node_id != 0) {
139 std::cout << "Waiting for raft cluster";
140 do {
141 srv_configs.clear();
142 m_raft_instance->get_srv_config_all(srv_configs);
143 std::cout << "." << std::flush;
144 std::this_thread::sleep_for(sleep_time);
145 } while(srv_configs.size() < raft_servers.size());
146 std::cout << "done" << std::endl;
147 } else if(!add_cluster_nodes(raft_servers)) {
148 return false;
149 }
150 }
151
152 m_raft_instance->restart_election_timer();
153
154 return true;
155 }

161 auto node::replicate(nuraft::ptr<nuraft::buffer> new_log,
162 const callback_type& result_fn) const -> bool {
163 auto ret = m_raft_instance->append_entries({std::move(new_log)});
164 if(!ret->get_accepted()) {
165 return false;
166 }
167
168 if(result_fn) {
169 ret->when_ready(result_fn);
170 }
171
172 return true;

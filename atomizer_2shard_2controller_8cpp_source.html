35 auto controller::init() -> bool {
36 if(auto err_msg
37 = m_shard.open_db(m_opts.m_shard_db_dirs[m_shard_id])) {
38 m_logger->error("Failed to open shard DB for shard",
39 m_shard_id,
40 ". Got error:",
41 *err_msg);
42 return false;
43 }
44
45 if(!m_archiver_client.init()) {
46 m_logger->error("Failed to connect to archiver");
47 return false;
48 }
49
50 if(!m_watchtower_network.cluster_connect(
51 m_opts.m_watchtower_internal_endpoints)) {
52 m_logger->error("Failed to connect to watchtowers.");
53 return false;
54 }
55
56 m_atomizer_network.cluster_connect(m_opts.m_atomizer_endpoints, false);
57 if(!m_atomizer_network.connected_to_one()) {
58 m_logger->error("Failed to connect to any atomizers");
59 return false;
60 }
61
62 m_atomizer_client = m_atomizer_network.start_handler([&](auto&& pkt) {
63 return atomizer_handler(std::forward<decltype(pkt)>(pkt));
64 });
65
66 auto ss = m_shard_network.start_server(
67 m_opts.m_shard_endpoints[m_shard_id],
68 [&](auto&& pkt) {
69 return server_handler(std::forward<decltype(pkt)>(pkt));
70 });
71
72 if(!ss.has_value()) {
73 m_logger->error("Failed to establish shard server.");
74 return false;
75 }
76
77 m_shard_server = std::move(ss.value());
78
79 return true;
80 }

82 auto controller::server_handler(cbdc::network::message_t&& pkt)
83 -> std::optional<cbdc::buffer> {
84 auto maybe_tx = from_buffer<transaction::compact_tx>(*pkt.m_pkt);
85 if(!maybe_tx.has_value()) {
86 m_logger->error("Invalid transaction packet");
87 return std::nullopt;
88 }
89
90 auto& tx = maybe_tx.value();
91
92 m_logger->info("Digesting transaction", to_string(tx.m_id), "...");
93
94 auto res = m_shard.digest_transaction(std::move(tx));
95
96 auto res_handler = overloaded{
97 [&](const atomizer::tx_notify_request& msg) {
98 m_logger->info("Digested transaction",
99 to_string(msg.m_tx.m_id));
100
101 m_logger->debug("Sending",
102 msg.m_attestations.size(),
103 "/",
104 msg.m_tx.m_inputs.size(),
105 "attestations...");
106 if(!m_atomizer_network.send_to_one(atomizer::request{msg})) {
107 m_logger->error("Failed to transmit tx to atomizer. ID:",
108 to_string(msg.m_tx.m_id));
109 }
110 },
111 [&](const cbdc::watchtower::tx_error& err) {
112 m_logger->info("error for Tx:",
113 to_string(err.tx_id()),
114 err.to_string());
115 // TODO: batch errors into a single RPC
116 auto data = std::vector<cbdc::watchtower::tx_error>{err};
117 auto buf = make_shared_buffer(data);
118 m_watchtower_network.broadcast(buf);
119 }};
120 std::visit(res_handler, res);
121 return std::nullopt;
122 }

124 auto controller::atomizer_handler(cbdc::network::message_t&& pkt)
125 -> std::optional<cbdc::buffer> {
126 auto maybe_blk = from_buffer<atomizer::block>(*pkt.m_pkt);
127 if(!maybe_blk.has_value()) {
128 m_logger->error("Invalid block packet");
129 return std::nullopt;
130 }
131
132 auto& blk = maybe_blk.value();
133
134 m_logger->info("Digesting block", blk.m_height, "...");
135
136 // If the block is not contiguous, catch up by requesting
137 // blocks from the archiver.
138 while(!m_shard.digest_block(blk)) {
139 m_logger->warn("Block", blk.m_height, "not contiguous.");
140
141 // Attempt to catch up to the latest block
142 for(uint64_t i = m_shard.best_block_height() + 1; i < blk.m_height;
143 i++) {
144 const auto past_blk = m_archiver_client.get_block(i);
145 if(past_blk) {
146 m_shard.digest_block(past_blk.value());
147 } else {
148 m_logger->info("Waiting for archiver sync");
149 const auto wait_time = std::chrono::milliseconds(10);
150 std::this_thread::sleep_for(wait_time);
151 i--;
152 continue;
153 }
154 }
155 }
156
157 m_logger->info("Digested block", blk.m_height);
158 return std::nullopt;
159 }

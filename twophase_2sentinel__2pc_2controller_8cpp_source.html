49 auto controller::execute_transaction(transaction::full_tx tx,
50 result_callback_type result_callback)
51 -> bool {
52 const auto validation_err = transaction::validation::check_tx(tx);
53 if(validation_err.has_value()) {
54 auto tx_id = transaction::tx_id(tx);
55 m_logger->debug(
56 "Rejected (",
57 transaction::validation::to_string(validation_err.value()),
58 ")",
59 to_string(tx_id));
60 result_callback(cbdc::sentinel::response{
61 cbdc::sentinel::tx_status::static_invalid,
62 validation_err});
63 return true;
64 }
65
66 auto compact_tx = cbdc::transaction::compact_tx(tx);
67
68 m_logger->debug("Accepted", to_string(compact_tx.m_id));
69
70 auto cb =
71 [&, res_cb = std::move(result_callback)](std::optional<bool> res) {
72 result_handler(res, res_cb);
73 };
74
75 // TODO: add a "retry" error response to offload sentinels from this
76 // infinite retry responsibility.
77 while(!m_coordinator_client.execute_transaction(compact_tx, cb)) {
78 // TODO: the network currently doesn't provide a callback for
79 // reconnection events so we have to sleep here to
80 // prevent a needless spin. Instead, add such a callback
81 // or queue to the network to remove this sleep.
82 static constexpr auto retry_delay = std::chrono::milliseconds(100);
83 std::this_thread::sleep_for(retry_delay);
84 };
85
86 return true;
87 }

89 void controller::result_handler(std::optional<bool> res,
90 const result_callback_type& res_cb) {
91 if(res.has_value()) {
92 auto resp = cbdc::sentinel::response{
93 cbdc::sentinel::tx_status::confirmed,
94 std::nullopt};
95 if(!res.value()) {
96 resp.m_tx_status = cbdc::sentinel::tx_status::state_invalid;
97 }
98 res_cb(resp);
99 } else {
100 res_cb(std::nullopt);
101 }
102 }
